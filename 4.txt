package com.example.wallaper;
import androidx.appcompat.app.AppCompatActivity;
import android.annotation.SuppressLint;
import android.app.WallpaperManager;
import android.graphics.Bitmap;
import android.graphics.drawable.BitmapDrawable;
import android.graphics.drawable.Drawable;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import java.io.IOException;
import java.util.Timer;
import java.util.TimerTask;
public class MainActivity extends AppCompatActivity {
Button btnChange;
Timer myTimer;
WallpaperManager myManager;
Drawable drawable;
int pre=1;
@Override
protected void onCreate(Bundle savedInstanceState) {
super.onCreate(savedInstanceState);
setContentView(R.layout.activity_main);
btnChange = findViewById(R.id.btnChange);
myManager = WallpaperManager.getInstance(this);
myTimer = new Timer();
btnChange.setOnClickListener(new View.OnClickListener() {
@Override
public void onClick(View view) {
setWallpaper();
}
});
}
public void setWallpaper()
{
myTimer.schedule(new TimerTask() {
@SuppressLint("MissingPermission")
@Override
public void run()
{
if(pre==1) {
drawable=getResources().getDrawable(R.drawable.pic1);
pre=2;
}
else if(pre==2) {
drawable=getResources().getDrawable(R.drawable.pic2);
pre=3;
}
else if(pre==3) {
drawable=getResources().getDrawable(R.drawable.pic3);
pre=4;
}
else if(pre==4) {
drawable=getResources().getDrawable(R.drawable.p4);
pre=5;
}
else if(pre==5) {
drawable=getResources().getDrawable(R.drawable.p5);
pre=6;
}
else if(pre==6) {
drawable=getResources().getDrawable(R.drawable.p6);
pre=7;
}
else if(pre==7) {
drawable=getResources().getDrawable(R.drawable.p7);
pre=8;
}
else if(pre==8) {
drawable=getResources().getDrawable(R.drawable.p8);
pre=1;
}
Bitmap wallpaper = ((BitmapDrawable)drawable).getBitmap();
try {
myManager.setBitmap(wallpaper);
} catch (IOException e) {
e.printStackTrace();
}
}
},0,3000);
}
}
